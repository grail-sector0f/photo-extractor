```typescript
export interface SavedPhotoRecord {
  id: string;          // timestamp-based unique ID, e.g. "1711234567890-0"
  url: string;         // original source URL (used for thumbnail display)
  filename: string;    // relative path: "travel-photos/bali_four-seasons_pool_2025_01.jpg"
  destination: string; // normalized destination field value
  vendor: string;      // normalized vendor field value
  category: string;    // category field value
  year: string;        // year field value
  notes: string;       // notes field value (may be empty string)
  savedAt: string;     // ISO 8601 timestamp
}

const STORAGE_KEY = 'savedPhotos';
const MAX_RECORDS = 500;

export async function loadLibrary(): Promise<SavedPhotoRecord[]> {
  const data = await chrome.storage.local.get(STORAGE_KEY);
  return (data[STORAGE_KEY] as SavedPhotoRecord[]) ?? [];
}

export async function appendToLibrary(record: SavedPhotoRecord): Promise<void> {
  const existing = await loadLibrary();
  const updated = [...existing, record];
  const capped = updated.length > MAX_RECORDS
    ? updated.slice(updated.length - MAX_RECORDS)
    : updated;
  await chrome.storage.local.set({ [STORAGE_KEY]: capped });
}
```

Export STORAGE_KEY and MAX_RECORDS as named exports so tests can verify values directly.
